41 // baseline release version.
42  E_SWM_PROT_Tag_remove = 0x06,
43 // Get the ECU source image path.
44  E_SWM_PROT_Tag_sourceDataPath = 0x07,
45 // Response only: get the percent progress of an operation.
46  E_SWM_PROT_Tag_progress = 0x08,
47 // Response only: get the result of an operation.
48  E_SWM_PROT_Tag_updateResult = 0x09,
49 // Get next ECU component ID: This request is intended to iterate over
50 // the IDs. For each request, return the next ID. Return NULL to indicate
51 // the end of the list of ECU components. This ID is not related to the
52 // ECU hardware ID below.
53  E_SWM_PROT_Tag_getCompId = 0X0A,
54 // Ask the ECU to release the structure sent with the previous request.
55  E_SWM_PROT_Tag_releaseMemory = 0X0B,
56 // Get the ECU hardware serial number.
57  E_SWM_PROT_Tag_getSerialNumber = 0x0C,
58 // Get the ECU hardware part number.
59  E_SWM_PROT_Tag_getPartNumber = 0x0D,
60 // Get the ECU hardware ID of the ECU’s management unit. If the
61 // ECU does not have a management unit, this must return null.
62  E_SWM_PROT_Tag_getManagementUnit = 0x0E,
63 // Get the ECU hardware address.
64  E_SWM_PROT_Tag_getNodeAddress = 0x0F,
65 // Ask the vehicle to begin collecting ECU data. Collecting hardware data
66 // (getSerialNumber, getPartNumber, getManagementUnit, getNodeAddress,
67 // and getEcuId) from the ECUs might take a long time (several minutes).
68 // To collect ECU data, the engine first sends this init request. The
69 // engine then sends the other requests for hardware data from each ECU.
70  E_SWM_PROT_Tag_init = 0x10,
71 // Not yet implemented
72  E_SWM_PROT_Tag_term = 0x11,
73 // Get the next ECU hardware ID: This request is intended to iterate over
74 // the IDs. For each request, return the next ID. Return NULL to indicate
75 // the end of the list of ECUs. The ECU hardware ID is either the node
76 // address (the location on the CAN bus) or any other string that can
77 // uniquely identify the ECU. This ID is not related to the ECU component
78 // ID above.
79  E_SWM_PROT_Tag_getEcuId = 0x12,
80 // Get current ECU name
81  E_SWM_PROT_Tag_getName = 0x13,
82 } E_SWM_PROT_Tag_t;
